### 全文索引  
增加bts vp 
```shell
onmode -p +1 bts
```
注：onmode 是临时增加，永久增加需要在onconfig中增加 VPCLASS bts,num=1

在需要使用bts功能的库中注册使用bts  
```shell
unset DB_LOCALE
unset CLIENT_LOCALE
blademgr
register bts.3.10 testdb
```

创建全文索引，及测试  
```sql
drop table if exists tvarchar;
create table tvarchar(col1 int, col2 varchar(255));
create index ix_tvarchar_col2 on tvarchar(col2 bts_varchar_ops) USING bts(analyzer="cjk");

insert into tvarchar values(1,'天津南大通用');
insert into tvarchar values(2,'南大通用北京太阳宫大厦');
insert into tvarchar values(3,'海泰发展六道6号');
insert into tvarchar values(4,'GENERAL DATA');
insert into tvarchar values(5,'AA BB CC DD');
insert into tvarchar values(6,'CC FF');

select * from tvarchar where bts_contains(col2,'南大');

select * from tvarchar where bts_contains(col2,'DATA');
```
