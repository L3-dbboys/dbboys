### 使用外部表导入导出
```
create raw table testtab(col1 serial,col2 int,col3 decimal,col4 char(100),col5 varchar(20),col6 datetime year to fraction(5));
insert into testtab values(0,1,1,'a','b',current);
SET ENVIRONMENT IFX_BATCHEDREAD_TABLE '1';
set pdqpriority 90;
insert into testtab_ext select * from testtab;
drop table testtab;
create raw table testtab(col1 serial,col2 int,col3 decimal,col4 char(100),col5 varchar(20),col6 datetime year to fraction(5));
insert into testtab select * from testtab_ext;
```

### 分区表
```
--范围分区
create table tes1(c1 int , c2 char(10), c3 date)
FRAGMENT BY EXPRESSION
c1 < 100 IN datadbs01,
c1 >= 100 AND c1 < 200 IN datadbs02,
c1 >= 200 IN datadbs03;

--轮转分区

create table test2(c1 int , c2 char(10))
partition by round robin  
partition p1 in datadbs01,
partition p2 in datadbs02,
partition p3 in datadbs03  ;

--列表分区

create table test3(c1 int, c2 char(20), c3 varchar(10),c4 date)
partition by list (c2) 
partition p11 values('aaaa','bbbb') in datadbs01,
partition p12 values('cccc','dddd') in datadbs02,
partition p13 values('eeee','ffff') in datadbs01,
partition p14 values('gggg','hhhh') in datadbs02
;  

--间隔分区
CREATE TABLE test4 (id INTEGER, na CHAR(32), dept_id CHAR(2), mgr_id INTEGER, ssn CHAR(12))
FRAGMENT BY RANGE (id) 
INTERVAL (100) STORE IN (datadbs1,datadbs2)
PARTITION p0 VALUES < 200 IN datadbs1;

--哈希分区
CREATE TABLE test5(
name varchar(20),
departmentID int,
age int,
address varchar(50)
)PARTITION BY HASH(departmentID)
(
partition p1 DBSPACE datadbs1,
partition p2 DBSPACE datadbs1,
partition p3 DBSPACE datadbs1
);

```
