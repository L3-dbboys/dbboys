### 导出
```shell
#unload.sh
#sh unload.sh <dbname>
export FET_BUF_SIZE=32767
DBNAME=$1
dbaccess -e -m ${DBNAME} -<<!
unload to unload.sql delimiter ";" select 'unload to ' ||tabname||'.unl select * from '||tabname from systables where tabid>99 and tabtype='T';
!
dbaccess -e -m ${DBNAME} unload.sql
```
### 导入
```shell
#load.sh
#sh load.sh <dbname>
export FET_BUF_SIZE=32767
DBNAME=$1
dbaccess -e -m ${DBNAME} -<<!
unload to load.sql delimiter ";" select 'load from ' ||tabname||'.unl insert into  '||tabname from systables where tabid>99 and tabtype='T';
!
dbaccess -e -m ${DBNAME} load.sql
```
